\begin{enumerate}
    \item Conmutatividad: $\vec{u} \cdot \vec{v} = \vec{v} \cdot \vec{u}$
    \item Asociatividad con escalar: $c(\vec{u} \cdot \vec{v}) = (c\vec{u}) \cdot \vec{v}$
\end{enumerate}
